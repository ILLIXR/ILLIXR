28inline Eigen::Matrix4d makeOmega(const Eigen::Vector3d& w) {

138state_plus predict_mean_rk4(double dt, const state_plus& sp, const Eigen::Vector3d& ang_vel, const Eigen::Vector3d& linear_acc,
139 const Eigen::Vector3d& ang_vel2, const Eigen::Vector3d& linear_acc2) {
140 Eigen::Vector3d av = ang_vel;
141 Eigen::Vector3d la = linear_acc;
142 const Eigen::Vector3d delta_av = (ang_vel2 - ang_vel) / dt;
143 const Eigen::Vector3d delta_la = (linear_acc2 - linear_acc) / dt;
144
145 // y0 ================
146 data_format::proper_quaterniond q_0 = sp.orientation; // initial orientation quaternion
147 Eigen::Vector3d p_0 = sp.position; // initial position vector
148 Eigen::Vector3d v_0 = sp.velocity; // initial velocity vector
149
150 // Calculate the RK4 coefficients
151 // solve orientation
152 // k1
153 data_format::proper_quaterniond k1_q = q_dot(av, dq_0) * dt;
154 av += 0.5 * delta_av * dt;
155 // k2
156 data_format::proper_quaterniond dq_1 = delta_q(k1_q);
157 data_format::proper_quaterniond k2_q = q_dot(av, dq_1) * dt;
158 // k3
159 data_format::proper_quaterniond dq_2 = delta_q(k2_q);
160 data_format::proper_quaterniond k3_q = q_dot(av, dq_2) * dt;
161 // k4
162 av += 0.5 * delta_av * dt;
163 data_format::proper_quaterniond dq_3 = delta_q(2. * k3_q);
164 data_format::proper_quaterniond k4_q = q_dot(av, dq_3) * dt;
165
166 // solve velocity
167 // k1
168 Eigen::Vector3d k1_v = v_dot(dq_0, q_0, la) * dt;
169 // k2
170 la += 0.5 * delta_la * dt;
171 Eigen::Vector3d k2_v = v_dot(dq_1, q_0, la) * dt;
172 // k3
173 Eigen::Vector3d k3_v = v_dot(dq_2, q_0, la) * dt;
174 // k4
175 la += 0.5 * delta_la * dt;
176 Eigen::Vector3d k4_v = v_dot(dq_3, q_0, la) * dt;
177
178 // solve position
179 // k1
180 Eigen::Vector3d k1_p = v_0 * dt;
181 // k2
182 Eigen::Vector3d k2_p = p_dot(v_0, k1_v) * dt;
183 // k3
184 Eigen::Vector3d k3_p = p_dot(v_0, k2_v) * dt;
185 // k4
186 Eigen::Vector3d k4_p = p_dot(v_0, 2. * k3_v) * dt;
187
188 // y+dt ================
189 state_plus state_plus;
190 data_format::proper_quaterniond dq = solve(dq_0, k1_q, k2_q, k3_q, k4_q);
191 dq.normalize();
192 state_plus.orientation = q_0 * dq;
193 state_plus.orientation.normalize();
194 state_plus.position = solve(p_0, k1_p, k2_p, k3_p, k4_p);
195 state_plus.velocity = solve(v_0, k1_v, k2_v, k3_v, k4_v);
196 return state_plus;
197}
