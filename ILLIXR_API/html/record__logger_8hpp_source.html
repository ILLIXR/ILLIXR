31 record_header(const std::string& name, std::vector<std::pair<std::string, const std::type_info&>> columns_)
32 : id_{std::hash<std::string>{}(name)}

39 bool operator==(const record_header& other) const {
40 // Check pointer first
41 if (this == &other) {
42 return true;
43 }
44
45 if (name_ != other.name_ || columns_.size() != other.columns_.size() || id_ != other.id_) {
46 return false;
47 }
48 for (std::size_t i = 0; i < columns_.size(); ++i) {
49 if (columns_[i] != other.columns_[i]) {
50 return false;
51 }
52 }
53 return true;
54 }

69 return columns_[column].first;
70 }
71
72 [[nodiscard]] const std::type_info& get_column_type(unsigned column) const {
73 return columns_[column].second;
74 }

77 return columns_.size();
78 }

81 std::string ret = std::string{"record_header "} + name_ + std::string{" { "};

145public:
146 record(const record_header& rh, std::vector<std::any> values)
147 : record_header_{rh}
148 , values_(std::move(values)) {
149#ifndef NDEBUG
150 assert(record_header_);
151 if (values_.size() != record_header_->get().get_columns()) {
152 spdlog::get("illixr")->error("[record_logger] {} elements passed, but rh for {} only specifies {}.", values_.size(),
153 record_header_->get().get_name(), record_header_->get().get_columns());
154 abort();
155 }
156 for (std::size_t column = 0; column < values_.size(); ++column) {
157 if (values_[column].type() != record_header_->get().get_column_type(column)) {
158 spdlog::get("illixr")->error("[record_logger] Caller got wrong type for column {} of {}.", column,
159 record_header_->get().get_name());
160 spdlog::get("illixr")->error("[record_logger] Caller passed: {}; record_header specifies: {}",
161 values_[column].type().name(),
162 record_header_->get().get_column_type(column).name());
163 abort();
164 }
165 }
166#endif
167 }

321 void log(const record& r) {
322 if (logger_) {
323 buffer_.push_back(r);
324 // Log coalescer should only be used with
325 // In the common case, they will be the same pointer, quickly check the pointers.
326 // In the less common case, we check for object-structural equality.
327#ifndef NDEBUG
328 if (&r.get_record_header() != &buffer_[0].get_record_header() &&
329 r.get_record_header() == buffer_[0].get_record_header()) {
330 spdlog::get("illixr")->error("[record_logger] Tried to push a record of type {} to a record logger for type {}",
331 r.get_record_header().to_string(), buffer_[0].get_record_header().to_string());
332 abort();
333 }
334#endif
335 maybe_flush();
336 }
337 }

342 void maybe_flush() {
343 if (std::chrono::high_resolution_clock::now() > last_log_ + LOG_BUFFER_DELAY) {
344 flush();
345 }
346 }
347
351 void flush() {
352 if (logger_) {
353 std::vector<record> buffer2;
354 buffer_.swap(buffer2);
355 logger_->log(buffer2);
356 last_log_ = std::chrono::high_resolution_clock::now();
357 }
358 }
359
360 explicit operator bool() const {
361 return bool(logger_);
362 }
363
364private:

366 std::chrono::time_point<std::chrono::high_resolution_clock> last_log_;
367 std::vector<record> buffer_;
