132 [[maybe_unused]] void flip_y(const uint im_height = 0) {
133 if (unit == units::PERCENT) {
134 for (auto& pnt : points) {
135 if (pnt.y() != 0.)
136 pnt.y() = 1.0f - pnt.y();
137 }
138 return;
139 }
140 if (im_height == 0)
141 throw std::runtime_error("Cannot rectify point with non-percent units with no height given.");
142 for (auto& pnt : points) {
143 if (pnt.y() != 0.)
144 pnt.y() = (float) im_height - pnt.y();
145 }

312 struct raw_ht_data {

314 raw_point h_points[2][NUM_LANDMARKS];

316 raw_pose wcs_origin;
317 coordinates::reference_space reference = coordinates::VIEWER;
318 units::measurement_unit unit = units::measurement_unit::UNSET;
319 coordinates::frame frame = coordinates::RIGHT_HANDED_Y_UP;

329
339 raw_ht_data(const time_point time, const std::map<ht::hand, hand_points>& points,
340 const std::map<ht::hand, velocity>& velocities, const pose_data& pose,
341 coordinates::reference_space ref_sp = coordinates::VIEWER,
342 units::measurement_unit _unit = units::measurement_unit::UNSET)
343 : time{static_cast<uint64_t>(time.time_since_epoch().count())}
344 , reference{ref_sp}
345 , unit{_unit}
346 , valid{true} {
347 for (auto i = 0; i < NUM_LANDMARKS; i++) {
348 h_points[LEFT_HAND][i].copy(points.at(LEFT_HAND).at(i));
349 hp_valid[LEFT_HAND] |= points.at(LEFT_HAND).at(i).valid;
350 h_points[RIGHT_HAND][i].copy(points.at(RIGHT_HAND).at(i));
351 hp_valid[RIGHT_HAND] |= points.at(RIGHT_HAND).at(i).valid;
352 h_velocities[LEFT_HAND][i].copy(velocities.at(LEFT_HAND).at(i));
353 hv_valid[LEFT_HAND] |= velocities.at(LEFT_HAND).at(i).valid;
354 h_velocities[RIGHT_HAND][i].copy(velocities.at(RIGHT_HAND).at(i));
355 hv_valid[RIGHT_HAND] |= velocities.at(RIGHT_HAND).at(i).valid;
356 }
357
358 wcs_origin.copy(pose);
359
360 if (unit == units::measurement_unit::UNSET) {
361 if (pose.valid) {
362 unit = pose.unit;
363 } else if (points.count(LEFT_HAND) > 0 && points.at(LEFT_HAND).valid) {
364 unit = points.at(LEFT_HAND).unit;
365 } else if (points.count(RIGHT_HAND) > 0 && points.at(RIGHT_HAND).valid) {
366 unit = points.at(RIGHT_HAND).unit;
367 }
368 }
369 }

383 void copy(const ht_frame& frame_) {
384 time = static_cast<uint64_t>(frame_.time.time_since_epoch().count());
385 reference = frame_.reference;
386 unit = frame_.unit;
387
388 for (auto i = 0; i < NUM_LANDMARKS; i++) {
389 h_points[LEFT_HAND][i].copy(frame_.hand_positions.at(LEFT_HAND).at(i));
390 hp_valid[LEFT_HAND] = hp_valid[LEFT_HAND] || frame_.hand_positions.at(LEFT_HAND).at(i).valid;
391 h_points[RIGHT_HAND][i].copy(frame_.hand_positions.at(RIGHT_HAND).at(i));
392 hp_valid[RIGHT_HAND] = hp_valid[RIGHT_HAND] || frame_.hand_positions.at(RIGHT_HAND).at(i).valid;
393 h_velocities[LEFT_HAND][i].copy(frame_.hand_velocities.at(LEFT_HAND).at(i));
394 hv_valid[LEFT_HAND] = hv_valid[LEFT_HAND] || frame_.hand_velocities.at(LEFT_HAND).at(i).valid;
395 h_velocities[RIGHT_HAND][i].copy(frame_.hand_velocities.at(RIGHT_HAND).at(i));
396 hv_valid[RIGHT_HAND] = hv_valid[RIGHT_HAND] || frame_.hand_velocities.at(RIGHT_HAND).at(i).valid;
397 }
398
399 wcs_origin.copy(frame_.wcs_offset);
400 }
