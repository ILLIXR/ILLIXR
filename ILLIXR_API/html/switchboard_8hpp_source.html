238 void thread_body() {
239 // Try to pull event off of queue
240 ptr<const event> this_event;
241 std::int64_t timeout_usecs = std::chrono::duration_cast<std::chrono::microseconds>(queue_timeout_).count();
242 // Note the use of timed blocking wait
243 if (queue_.wait_dequeue_timed(token_, this_event, timeout_usecs)) {
244 // Process event
245 // Also, record and log the time
246 dequeued_++;
247 auto cb_start_cpu_time = thread_cpu_time();
248 auto cb_start_wall_time = std::chrono::high_resolution_clock::now();
249 // std::cerr << "deq " << ptr_to_str(reinterpret_cast<const void*>(this_event.get_ro())) << " " <<
250 // this_event.use_count() << " v\n";
251 callback_(std::move(this_event), dequeued_);
252 if (cb_log_) {
253 cb_log_.log(record{_switchboard_callback_header,
254 {
255 {plugin_id_},
256 {topic_name_},
257 {dequeued_},
258 {cb_start_cpu_time},
259 {thread_cpu_time()},
260 {cb_start_wall_time},
261 {std::chrono::high_resolution_clock::now()},
262 }});
263 }

271 // Drain queue

274 ptr<const event> this_event;
