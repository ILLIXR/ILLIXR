175  void thread_body() {
176  // Try to pull event off of queue
177  ptr<const event> this_event;
178  std::int64_t timeout_usecs = std::chrono::duration_cast<std::chrono::microseconds>(_m_queue_timeout).count();
179  // Note the use of timed blocking wait
180  if (_m_queue.wait_dequeue_timed(_m_ctok, this_event, timeout_usecs)) {
181  // Process event
182  // Also, record and log the time
183  _m_dequeued++;
184  auto cb_start_cpu_time = thread_cpu_time();
185  auto cb_start_wall_time = std::chrono::high_resolution_clock::now();
186  // std::cerr << "deq " << ptr_to_str(reinterpret_cast<const void*>(this_event.get_ro())) << " " << this_event.use_count() << " v\n";
187  _m_callback(std::move(this_event), _m_dequeued);
188  if (_m_cb_log) {
189  _m_cb_log.log(record{__switchboard_callback_header, {
190  {_m_plugin_id},
191  {_m_topic_name},
192  {_m_dequeued},
193  {cb_start_cpu_time},
194  {thread_cpu_time()},
195  {cb_start_wall_time},
196  {std::chrono::high_resolution_clock::now()},
197  }});
198  }
199  } else {
200  // Nothing to do.
201  _m_idle_cycles++;
202  }
203  }

206  void thread_on_stop() {
207  // Drain queue
208  std::size_t unprocessed = _m_enqueued - _m_dequeued;
209  {
210  ptr<const event> this_event;
211  for (std::size_t i = 0; i < unprocessed; ++i) {
212  [[maybe_unused]] bool ret = _m_queue.try_dequeue(_m_ctok, this_event);
213  assert(ret);
214  // std::cerr << "deq (stopping) " << ptr_to_str(reinterpret_cast<const void*>(this_event.get_ro())) << " " << this_event.use_count() << " v\n";
215  this_event.reset();
216  }
217  }
218 
219  // Log stats
220  if (_m_record_logger) {
221  _m_record_logger->log(record{__switchboard_topic_stop_header, {
222  {_m_plugin_id},
223  {_m_topic_name},
224  {_m_dequeued},
225  {unprocessed},
226  {_m_idle_cycles},
227  }});

326  void put(ptr<const event>&& this_event) {
327  assert(this_event != nullptr);
328  assert(this_event.unique() || this_event.use_count() <= 2);
329 
330  /* The pointer that this gets exchanged with needs to get dropped. */
331  size_t index = (_m_latest_index.load() + 1) % _m_latest_buffer_size;
332  _m_latest_buffer[index] = this_event;
333  _m_latest_index++;
334 
335  // Read/write on _m_subscriptions.
336  // Must acquire shared state on _m_subscriptions_lock
337  std::unique_lock lock{_m_subscriptions_lock};
338  for (topic_subscription& ts : _m_subscriptions) {
339  // std::cerr << "enq " << ptr_to_str(reinterpret_cast<const void*>(this_event->get())) << " " << this_event->use_count() << " ^\n";
340  ptr<const event> event_ptr_copy {this_event};
341  ts.enqueue(std::move(event_ptr_copy));
342  }
343  // std::cerr << "put done " << ptr_to_str(reinterpret_cast<const void*>(this_event->get())) << " " << this_event->use_count() << " (= 1 + len(sub)) \n";
344  }
