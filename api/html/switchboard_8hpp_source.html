179  void thread_body() {
180  // Try to pull event off of queue
181  ptr<const event> this_event;
182  std::int64_t timeout_usecs = std::chrono::duration_cast<std::chrono::microseconds>(_m_queue_timeout).count();
183  // Note the use of timed blocking wait
184  if (_m_queue.wait_dequeue_timed(_m_ctok, this_event, timeout_usecs)) {
185  // Process event
186  // Also, record and log the time
187  _m_dequeued++;
188  auto cb_start_cpu_time = thread_cpu_time();
189  auto cb_start_wall_time = std::chrono::high_resolution_clock::now();
190  // std::cerr << "deq " << ptr_to_str(reinterpret_cast<const void*>(this_event.get_ro())) << " " << this_event.use_count() << " v\n";
191  _m_callback(std::move(this_event), _m_dequeued);
192  if (_m_cb_log) {
193  _m_cb_log.log(record{__switchboard_callback_header, {
194  {_m_plugin_id},
195  {_m_topic_name},
196  {_m_dequeued},
197  {cb_start_cpu_time},
198  {thread_cpu_time()},
199  {cb_start_wall_time},
200  {std::chrono::high_resolution_clock::now()},
201  }});
202  }
203  } else {
204  // Nothing to do.
205  _m_idle_cycles++;
206  }
207  }

210  void thread_on_stop() {
211  // Drain queue
212  std::size_t unprocessed = _m_enqueued - _m_dequeued;
213  {
214  ptr<const event> this_event;
215  for (std::size_t i = 0; i < unprocessed; ++i) {
216  [[maybe_unused]] bool ret = _m_queue.try_dequeue(_m_ctok, this_event);
217  assert(ret);
218  // std::cerr << "deq (stopping) " << ptr_to_str(reinterpret_cast<const void*>(this_event.get_ro())) << " " << this_event.use_count() << " v\n";
219  this_event.reset();
220  }
221  }
222 
223  // Log stats
224  if (_m_record_logger) {
225  _m_record_logger->log(record{__switchboard_topic_stop_header, {
226  {_m_plugin_id},
227  {_m_topic_name},
228  {_m_dequeued},
229  {unprocessed},
230  {_m_idle_cycles},
231  }});

330  void put(ptr<const event>&& this_event) {
331  assert(this_event != nullptr);
332  assert(this_event.unique() || this_event.use_count() <= 2);
333 
334  /* The pointer that this gets exchanged with needs to get dropped. */
335  size_t index = (_m_latest_index.load() + 1) % _m_latest_buffer_size;
336  _m_latest_buffer[index] = this_event;
337  _m_latest_index++;
338 
339  // Read/write on _m_subscriptions.
340  // Must acquire shared state on _m_subscriptions_lock
341  std::unique_lock lock{_m_subscriptions_lock};
342  for (topic_subscription& ts : _m_subscriptions) {
343  // std::cerr << "enq " << ptr_to_str(reinterpret_cast<const void*>(this_event->get())) << " " << this_event->use_count() << " ^\n";
344  ptr<const event> event_ptr_copy {this_event};
345  ts.enqueue(std::move(event_ptr_copy));
346  }
