126 void process(const std::vector<record>& record_batch, std::size_t batch_size) {
127 sqlite3pp::transaction xct{database_};
128 for (std::size_t i = 0; i < batch_size; ++i) {
129 // TODO(performance): reuse the sqlite3pp statement
130 // This currently has to be 'reinterpreted' for every iteration.
131 sqlite3pp::command cmd{database_, insert_str_.c_str()};
132 const record& r = record_batch[i];
133 for (int j = 0; j < static_cast<int>(record_header_.get_columns()); ++j) {
134 /*
135 If you get a `std::bad_any_cast` here, make sure the user didn't lie about record.get_record_header().
136 The types there should be the same as those in record.get_values().
137 */
138 const std::type_info& rh_type = record_header_.get_column_type(j);
139 if (rh_type == typeid(std::size_t)) {
140 cmd.bind(j + 1, static_cast<long long>(r.get_value<std::size_t>(j)));
141 } else if (rh_type == typeid(bool)) {
142 cmd.bind(j + 1, static_cast<long long>(r.get_value<bool>(j)));
143 } else if (rh_type == typeid(double)) {
144 cmd.bind(j + 1, r.get_value<double>(j));
145 } else if (rh_type == typeid(std::chrono::nanoseconds)) {
146 auto val = r.get_value<duration>(j);
147 cmd.bind(j + 1, static_cast<long long>(std::chrono::nanoseconds{val}.count()));
148 } else if (rh_type == typeid(std::chrono::high_resolution_clock::time_point)) {
149 auto val = r.get_value<std::chrono::high_resolution_clock::time_point>(j).time_since_epoch();
150 cmd.bind(j + 1, static_cast<long long>(std::chrono::nanoseconds{val}.count()));
151 } else if (rh_type == typeid(duration)) {
152 auto val = r.get_value<duration>(j);
153 cmd.bind(j + 1, static_cast<long long>(std::chrono::nanoseconds{val}.count()));
154 } else if (rh_type == typeid(time_point)) {
155 auto val = r.get_value<time_point>(j).time_since_epoch();
156 cmd.bind(j + 1, static_cast<long long>(std::chrono::nanoseconds{val}.count()));
157 } else if (rh_type == typeid(std::string)) {
158 // r.get_value<std::string>(j) returns a std::string temporary
159 // c_str() returns a pointer into that std::string temporary
160 // Therefore, need to copy.
161 cmd.bind(j + 1, r.get_value<std::string>(j).c_str(), sqlite3pp::copy);
162 } else {
163 throw std::runtime_error{std::string{"type "} + std::string{record_header_.get_column_type(j).name()} +
164 std::string{" not implemented"}};
165 }
166 }
167 RAC_ERRNO_MSG("sqlite_record_logger set errno before process cmd execute");
168
169 cmd.execute();
170 RAC_ERRNO_MSG("sqlite_record_logger after process cmd execute");
171 }
172 xct.commit();
173 }
