13void projection(Eigen::Matrix4f* result, const float tan_left, const float tan_right, const float tan_up, float const tan_down,
14 const float near_z, const float far_z) {
15 const float tan_width = tan_right - tan_left;
16 const float tan_height = tan_up - tan_down;
17
18 // https://www.scratchapixel.com/lessons/3d-basic-rendering/perspective-and-orthographic-projection-matrix/building-basic-perspective-projection-matrix
19 (*result)(0, 0) = 2 / tan_width;
20 (*result)(0, 1) = 0;
21 (*result)(0, 2) = (tan_right + tan_left) / tan_width;
22 (*result)(0, 3) = 0;
23
24 (*result)(1, 0) = 0;
25 (*result)(1, 1) = 2 / tan_height;
26 (*result)(1, 2) = (tan_up + tan_down) / tan_height;
27 (*result)(1, 3) = 0;
28
29 (*result)(2, 0) = 0;
30 (*result)(2, 1) = 0;
31 (*result)(2, 2) = -far_z / (far_z - near_z);
32 (*result)(2, 3) = -(far_z * near_z) / (far_z - near_z);
33
34 (*result)(3, 0) = 0;
35 (*result)(3, 1) = 0;
36 (*result)(3, 2) = -1;
37 (*result)(3, 3) = 0;
38}

69void projection_fov(Eigen::Matrix4f* result, const float fov_left, const float fov_right, const float fov_up,

85void unreal_projection(Eigen::Matrix4f* result, const float fov_left, const float fov_right, const float fov_up,
86 const float fov_down) {
87 // Unreal uses a far plane at infinity and a near plane of 10 centimeters (0.1 meters)
88 constexpr float near_z = 0.1;
89
90 const float angle_left = tanf(static_cast<float>(fov_left));
91 const float angle_right = tanf(static_cast<float>(fov_right));
92 const float angle_up = tanf(static_cast<float>(fov_up));
93 const float angle_down = tanf(static_cast<float>(fov_down));
94
95 const float sum_rl = angle_left + angle_right;
96 const float sum_tb = angle_up + angle_down;
97 const float inv_rl = 1.0f / (angle_right - angle_left);
98 const float inv_tb = 1.0f / (angle_up - angle_down);
99
100 (*result)(0, 0) = 2 * inv_rl;
101 (*result)(0, 1) = 0;
102 (*result)(0, 2) = sum_rl * (-inv_rl);
103 (*result)(0, 3) = 0;
104
105 (*result)(1, 0) = 0;
106 (*result)(1, 1) = 2 * inv_tb;
107 (*result)(1, 2) = sum_tb * (-inv_tb);
108 (*result)(1, 3) = 0;
109
110 (*result)(2, 0) = 0;
111 (*result)(2, 1) = 0;
112 (*result)(2, 2) = 0;
113 (*result)(2, 3) = near_z;
114
115 (*result)(3, 0) = 0;
116 (*result)(3, 1) = 0;
117 (*result)(3, 2) = -1;
118 (*result)(3, 3) = 0;
119}

122[[maybe_unused]] void godot_projection(Eigen::Matrix4f* result, const float fov_left, const float fov_right, const float fov_up,
123 const float fov_down) {
124 // Godot's default far and near planes are 4000m and 0.05m respectively.
125 // https://github.com/godotengine/godot/blob/e96ad5af98547df71b50c4c4695ac348638113e0/modules/openxr/openxr_util.cpp#L97
126 // The Vulkan implementation passes in GRAPHICS_OPENGL for some reason.
127 constexpr float near_z = 0.05;
128 constexpr float far_z = 4000;
129 constexpr float offset_z = near_z;
130
131 const float angle_left = tanf(static_cast<float>(fov_left));
132 const float angle_right = tanf(static_cast<float>(fov_right));
133 const float angle_up = tanf(static_cast<float>(fov_up));
134 const float angle_down = tanf(static_cast<float>(fov_down));
135
136 const float angle_width = angle_right - angle_left;
137 const float angle_height = angle_up - angle_down;
138
139 Eigen::Matrix4f openxr_matrix;
140
141 openxr_matrix(0, 0) = 2 / angle_width;
142 openxr_matrix(0, 1) = 0;
143 openxr_matrix(0, 2) = (angle_right + angle_left) / angle_width;
144 openxr_matrix(0, 3) = 0;
145
146 openxr_matrix(1, 0) = 0;
147 openxr_matrix(1, 1) = 2 / angle_height;
148 openxr_matrix(1, 2) = (angle_up + angle_down) / angle_height;
149 openxr_matrix(1, 3) = 0;
150
151 openxr_matrix(2, 0) = 0;
152 openxr_matrix(2, 1) = 0;
153 openxr_matrix(2, 2) = -(far_z + offset_z) / (far_z - near_z);
154 openxr_matrix(2, 3) = -(far_z * (near_z + offset_z)) / (far_z - near_z);
155
156 openxr_matrix(3, 0) = 0;
157 openxr_matrix(3, 1) = 0;
158 openxr_matrix(3, 2) = -1;
159 openxr_matrix(3, 3) = 0;
160
161 // Godot then remaps the matrix...
162 // https://github.com/Khasehemwy/godot/blob/d950f5f83819240771aebb602bfdd4875363edce/core/math/projection.cpp#L722
163 Eigen::Matrix4f remap_z;
164
165 remap_z(0, 0) = 1;
166 remap_z(0, 1) = 0;
167 remap_z(0, 2) = 0;
168 remap_z(0, 3) = 0;
169
170 remap_z(1, 0) = 0;
171 remap_z(1, 1) = 1;
172 remap_z(1, 2) = 0;
173 remap_z(1, 3) = 0;
174
175 remap_z(2, 0) = 0;
176 remap_z(2, 1) = 0;
177 remap_z(2, 2) = -0.5;
178 remap_z(2, 3) = 0.5;
179
180 remap_z(3, 0) = 0;
181 remap_z(3, 1) = 0;
182 remap_z(3, 2) = 0;
183 remap_z(3, 3) = 1;
184
185 (*result) = remap_z * openxr_matrix;
186}

188/*
189 * Rotation matrix to convert a point from one coordinate system to another, e.g. left hand y up to right hand y up
190 *
191 */
192inline Eigen::Matrix3f rotation(const float alpha, const float beta, const float gamma) {
193 Eigen::Matrix3f rot;
194 double ra = alpha * M_PI / 180.;
195 double rb = beta * M_PI / 180.;
196 double rg = gamma * M_PI / 180;
197 rot << static_cast<float>(cos(rg) * cos(rb)), static_cast<float>(cos(rg) * sin(rb) * sin(ra) - sin(rg) * cos(ra)),
198 static_cast<float>(cos(rg) * sin(rb) * cos(ra) + sin(rg) * sin(ra)), static_cast<float>(sin(rg) * cos(rb)),
199 static_cast<float>(sin(rg) * sin(rb) * sin(ra) + cos(rg) * cos(ra)),
200 static_cast<float>(sin(rg) * sin(rb) * cos(ra) - cos(rg) * sin(ra)), static_cast<float>(-sin(rb)),
201 static_cast<float>(cos(rb) * sin(ra)), static_cast<float>(cos(rb) * cos(ra));
202 return rot;
203}

211// (right hand z up LHZU (left hand z up) RHZUXF (right hand z up x forward) to:
212const Eigen::Matrix3f conversion[6][6] = {
213 {identity, invert_y, rotation(180., 0., 0.), rotation(-90., 0., 0.), rotation(-90., 0., 90.) * invert_z,
214 rotation(-90., 0., -90.)}, // IM
215 {invert_y, identity, invert_z, rotation(-90., 0., 0.) * invert_y, rotation(90., 0., 90.),
216 rotation(-90., 0., -90.) * invert_y}, // LHYU
217 {rotation(180., 0., 0.), invert_z, identity, rotation(90., 0., 0.), rotation(90., 0., 90.) * invert_z,
218 rotation(-90., 0., 90.) * invert_x* invert_y}, // RHYU
219 {rotation(90., 0., 0.), invert_y* rotation(90., 0., 0.), rotation(-90., 0., 0.), identity, rotation(0., 0., 90.) * invert_y,
